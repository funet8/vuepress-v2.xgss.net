### 测试生成图片

```
import requests

# =========配置区========
API_KEY = "填写你的密钥"
BASE_URL = "https://apihub.agnes-ai.com/v1"  # 官方正确域名
PROMPT = "海边日落，写实摄影，8K高清，暖色调光影"
NEGATIVE_PROMPT = "模糊、畸形、低画质、水印"
IMAGE_SIZE = "1024x1024"
MODEL_NAME = "agnes-image-2.1-flash" # 文生图必填模型名
# ======================

def agnes_gen_img():
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": MODEL_NAME,
        "prompt": PROMPT,
        "negative_prompt": NEGATIVE_PROMPT,
        "size": IMAGE_SIZE,
        "n": 1
    }
    try:
        print("正在生成图片...")
        res = requests.post(f"{BASE_URL}/images/generations", headers=headers, json=payload, timeout=90)
        res.raise_for_status()
        json_data = res.json()
        img_url = json_data["data"][0]["url"]
        print(f"✅生成成功！图片链接：{img_url}")
        return img_url
    except Exception as err:
        print(f"❌失败：{err}")

if __name__ == "__main__":
    agnes_gen_img()
```
